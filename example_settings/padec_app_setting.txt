# This configuration file adds PADEC application for all the nodes

# Define new application
padecAppPro.type = PADECApp
padecAppPro.interval = 500
padecAppPro.provider = true

padecAppCon.type = PADECApp
padecAppCon.interval = 500
padecAppCon.destinationRange = 0,0
padecAppCon.consumer = true

# long range, interface
padecInterface.type = SimpleBroadcastInterface

# Define 2 different node groups
Scenario.nrofHostGroups = 2

# Group-specific settings:
Group.nrofHosts = 1
Group.nrofInterfaces = 1

# group1 (provider) specific settings
Group1.groupID = p
# group2 (consumer) specific settings
Group2.groupID = c

# Set PADEC app for all nodes
Group.nrofApplications = 1
Group1.application1 = padecAppPro
Group2.application1 = padecAppCon

# Message creation parameters
Events1.tohosts = 1, 1
Events1.hosts = 0, 1

# Add report for PADEC app
Report.nrofReports = 1
